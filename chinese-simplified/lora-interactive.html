class LoRALinear(nn.Module): def __init__(self): # 冻结的基础权重 self.W = nn.Linear(4096, 4096, bias=False) self.W.weight.requires_grad = False # 低秩适配器 —— 唯一可训练的参数 self.A = nn.Parameter(torch.randn(8, 4096) * 0.01) self.B = nn.Parameter(torch.zeros(4096, 8)) def forward(self, x): # x: (batch, seq, k) = (batch, seq, 4096) out = self.W(x) # → (batch, seq, 4096) lora = (x @ self.A.T) @ self.B.T # → (batch, seq, 8) → (batch, seq, 4096) return out + lora

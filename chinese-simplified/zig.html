Hello, Zig

一门小巧、简洁的语言，专为编写健壮、最优、可复用的软件而设计。无隐藏控制流，无隐式分配。

规范程序

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

注意返回类型 !void —— main 可以返回错误。try 会向上传播。

编译与运行 terminal
# 编译并运行单个文件
zig run main.zig

# 仅编译为原生可执行文件
zig build-exe main.zig

# 项目（使用 build.zig）
zig init
zig build run
zig build test

# 格式化代码
zig fmt main.zig

Zig 的构建系统由 Zig 自身写成 —— 不需要 Make，也不需要 CMake。

注释与测试 // /// test
// 普通注释
/// 文档注释（附着在下一个声明上）
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

测试与源码同处一文件。用 zig test file.zig 运行。

Zen of Zig philosophy
// 1. 精确传达意图。
// 2. 边界情形很重要。
// 3. 阅读代码胜于编写代码。
// 4. 做事只有一种显而易见的方式。
// 5. 运行时崩溃好过潜伏的 bug。
// 6. 编译错误好过运行时崩溃。
// 7. 渐进式改进。
// 8. 避免局部最优。
// 9. 减少必须记住的事情。
// 10. 关注代码而非风格。
// 11. 资源分配可能失败；释放必须成功。
// 12. 内存是一种资源。
// 13. 我们共同服务于用户。

变量

两种风味：const（不可变）与 var（可变）。所有声明都必须被使用 —— 不允许死绑定。

const 与 var basics
const name = "Ada";        // 不可变，类型推断
var   counter: u32 = 0;  // 可变，显式类型
counter += 1;

// 编译错误：不能给 const 赋值
// name = "Bob";

// 编译错误：未使用的局部变量
// const x = 5;

如果实在需要丢弃，用 _ = value;。

comptime 常量 comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// 编译期求值
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

顶层 const 隐式即为 comptime。

undefined 主动选择的不安全
var buf: [256]u8 = undefined;
// 在 debug 构建中，undefined 是 0xAA 字节，
// 这样能让未初始化的读取暴露出来。
buf[0] = 42;
类型推断 @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

数值与类型

Zig 提供显式、定宽的整数类型，并暴露溢出语义。所有类型转换都必须显式书写。

整数类型

类型位宽范围
i8 / u88-128..127 / 0..255
i16 / u1616有符号 / 无符号
i32 / u3232最常用的整数
i64 / u6464大整数
isize / usize指针指针宽度；索引一律用 usize
i7 / u23任意任意位宽 —— u3 也合法！
f16 / f32 / f64 / f128不一IEEE 浮点

类型转换与溢出

类型转换 @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // 运行时检查
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as 用于强制转换（拓宽、编译期）
const e = @as(i64, 10);

在安全构建中，@intCast 在值不能容纳时会触发陷阱（trap）。

溢出运算符 +% +| etc
// 默认 + 在安全构建中遇溢出会陷入
var x: u8 = 250;

// 回绕（wrapping）算术
const w = x +% 10;   // 回绕到 4

// 饱和（saturating）算术
const s = x +| 10;   // 钉在 255

// 用 @addWithOverflow 捕获溢出
const r = @addWithOverflow(x, 10);
// r 是元组：{ result, did_overflow }
数字字面量 readability
const million = 1_000_000;
const hex     = 0xff_ee;
const oct     = 0o755;
const bin     = 0b1010_1010;
const exp     = 1.5e3;
const hexf    = 0x1.fp10;  // 十六进制浮点
字符与字节 u8 'A'
const letter: u8 = 'A';       // 65
const emoji  : u21 = '😀';     // 码点

// 字符串是 []const u8（UTF-8 字节）
const greeting: []const u8 = "hi";

控制流

到处都是表达式。if、switch、代码块都会求值出一个值。switch 必须穷举。

if 当作表达式 if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// 可选类型的负载绑定
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// 配合可选类型的 while
while (it.next()) |item| {
    process(item);
}
for —— 现代语法 for
const items = [_]u8{ 10, 20, 30 };

// 遍历元素
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// 区间语法（Zig 0.11+）
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// 多参 for：值与索引一起拿
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// 对枚举做 switch（穷举即可，无需 else）
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
带标签的代码块 & 带值 break :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// 跳出嵌套循环
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer cleanup
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // 离开作用域时执行

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // 仅在出错路径上执行

try initialize(buf);
return buf;          // 成功时不触发 errdefer

函数

参数不可变。错误通过错误联合返回类型 !T 传递。

基础函数 fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// 参数都是 const —— 不能给 a 或 b 赋值
错误联合返回 !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // 向上传播
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
用结构体多返回值 .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
函数指针 *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

错误

错误是值，但是一种特殊的值。它们组成集合，并通过 try 传播。

错误集合 error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

错误集合是若干命名错误的并集。!T 使用推断出的集合。

try 与 catch control
// try：解包，否则把错误向上传
const n = try parse(input);

// catch：用一个值替代
const m = parse(input) catch 0;

// catch 配合负载绑定
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer 出错时清理
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // 这里失败的话 errdefer 会执行
    return t;               // 成功：errdefer 跳过
}
unreachable 断言
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // 在安全构建中陷入
}

// "我知道这里 parse 一定成功"
const n = parse("42") catch unreachable;

内存与分配器

没有隐藏的分配。任何会分配的函数都会显式接收一个 Allocator。你分配什么就释放什么（或者用 arena）。

分配器模式

GeneralPurposeAllocator 便于调试
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA 能检测内存泄漏与重复释放。在应用与测试里都建议用它。

ArenaAllocator 批量释放
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // 一次性把全部释放
const alloc = arena.allocator();

// 随便分配 —— 永远不要调用 free()
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
其他分配器 按需选择
// std.heap.page_allocator       —— 基于 mmap
// std.heap.c_allocator          —— libc malloc
// std.testing.allocator         —— 测试专用
// std.heap.FixedBufferAllocator —— 由栈缓冲支撑

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
数组 / 切片 / 指针 [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // 定长
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // 指针 + 长度
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // 单元素指针
const many: [*]const u8 = arr[0..].ptr; // 多元素指针

// 切片自带长度：arr.len

结构体与枚举

  
结构体既能装数据也能挂方法。枚举可以带载荷（即"带标签的联合体"）。两者都是一等类型。


  

    

      

        
结构体（带方法）struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });


    


    

      

        
枚举 enum

        
      

      
const Direction = enum { north, south, east, west };

// 指定底层类型并显式赋值
const Status = enum(u8) {
    ok = 200,
    not_found = 404,
    server_error = 500,
};

const code = @intFromEnum(Status.ok); // 200


    


    

      

        
带标签的联合体 union(enum)

        
      

      
const Token = union(enum) {
    int: i64,
    float: f64,
    text: []const u8,
    eof,
};

const t: Token = .{ .int = 42 };

switch (t) {
    .int => |n| std.debug.print("int {d}\n", .{n}),
    .float => |f| std.debug.print("float {d}\n", .{f}),
    .text => |s| std.debug.print("text {s}\n", .{s}),
    .eof => std.debug.print("end\n", .{}),
}

Comptime

  
Zig 的杀手锏。在编译期跑代码、把类型当参数、用普通 Zig 语法生成代码 —— 全部融为一体。


  

    

      

        
通过编译期类型实现泛型 comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
每次调用都会生成一个特化版本 —— 运行时无额外开销。

    


    

      

        
泛型结构体 返回 type 的函数

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for 展开循环

        
      

      
// 编译期把循环体按迭代次数展开
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// 也支持区间形式
inline for (0..4) |i| {
    process(i); // 展开 4 次
}


    


    

      

        
编译期执行代码 comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// 在编译期算出来：
const tenth = comptime fib(10);  // 55，烘焙进二进制

构建系统

  
build.zig 本身就是 Zig 代码。你描述出一张构建步骤的图，编译器负责执行。


  

    

      

        
最小 build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

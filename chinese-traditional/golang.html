變數與型別

Go 是靜態型別,但因為型別推斷與 short-declaration 語法,寫起來比 Java 輕。

宣告

var ok = true             // 推斷型別

// short form(只能在函式內)

age, ok := 42, true

// 區塊
var (
    host = "localhost"
    port = 8080
)
常數

const (

控制流

一種迴圈、一種 if、一種 switch — 但每個都份量十足。

    return err
}

// 帶 init 的 if;變數只在這個 if 內有效
if v, ok := m["key"]; ok {
    fmt.Println(v)
} else {
    fmt.Println("missing")
}
for — Go 唯一的迴圈
// 經典三段式
for i := 0; i < 10; i++ { … }

// while 風格

// 無限迴圈

// range
for i, v := range slice { … }

for i := range 10 { … }   // Go 1.22+
switch(預設沒有 fallthrough)
switch os := runtime.GOOS; os {
case "darwin":
    fmt.Println("mac")
case "linux", "freebsd":
    fmt.Println("unix")
default:
    fmt.Println(os)

switch {
case n < 0:   return "neg"
case n == 0:  return "zero"
default:      return "pos"

type switch
switch v := x.(type) {
case int:
    fmt.Println("int", v*2)
case string:
    fmt.Println("str", len(v))
case nil:
    fmt.Println("nil")
default:
    fmt.Printf("%T\n", v)
}
標籤、break、continue、goto
outer:
for _, row := range grid {
    for _, cell := range row {
        if cell == target {
            break outer
        }
    }
}
陷阱(1.22 之前):舊版 Go 的迴圈變數會被跨 iteration 重用 — 在 closure 或 goroutine 裡捕捉它,每次都綁到同一個位址。Go 1.22 修正了這點,每個 iteration 都有新的 v。如果還在用舊版 Go,請在迴圈裡先寫 v := v。

多重返回、命名返回、變動參數、defer、closure。函式就是值。

基本與多重返回

func divide(a, b float64) (float64, error) {
    if b == 0 {

    return a / b, nil

q, err := divide(10, 3)

defer
在函式 return 時執行。LIFO 順序。註冊 defer 當下參數就會被求值。
func readFile(path string) (string, error) {
    f, err := os.Open(path)
    if err != nil { return "", err }
    defer f.關閉()    // 離開函式時必定執行

    b, err := io.ReadAll(f)
    return string(b), err
}

    n := 0
    return func() int {
        n++
        return n
    }
}

next := counter()
fmt.Println(next(), next(), next())  // 1 2 3

指標

Go 有指標,但沒有指標算術。指標的兩個用途:避免複製,以及讓被呼叫者修改呼叫者的值。

取址、解參
x := 42
p := &x         // 指向 x 的指標
fmt.Println(*p)  // 42
*p = 100
fmt.Println(x)   // 100
傳值 vs 傳指標
func grow(n int)    { n *= 2 }      // 沒效
func growP(n *int) { *n *= 2 }    // 修改呼叫者

x := 3
grow(x)        // x 仍是 3
growP(&x)      // x 變成 6
new vs &T{}
// new(T) → 指向零值 T 的 *T
p := new(int)        // *int 指向 0
*p = 7

// 複合字面值 — 更慣用
u := &User{Name: "jane"}
何時該用指標
用指標:修改、大型結構(避免複製)、用指標 receiver 滿足介面、可選/nil 哨兵。
避免:小型不可變結構;slice/map/channel(本身就是參考型別)。
陷阱:舊版 Go 對 range 變數取址,每個 iteration 都是同一個位址。Go 1.22+ 每個 iteration 都有新變數,所以 range 中的 &v 安全。在舊版上請先 v := v。

結構與方法

結構裝資料。方法是有 receiver 的函式。Embedding 給你組合。

定義與建立
type User struct {
    ID    int      `json:"id"`
    Name  string   `json:"name"`
    Email string   `json:"email,omitempty"`
}

u := User{ID: 1, Name: "jane"}
v := User{1, "jane", ""}   // 位置式,易壞
p := &User{Name: "alex"}     // 指向新結構的指標
方法(值 receiver vs 指標 receiver)
type Counter struct{ n int }

// 值 receiver — 對副本操作
func (c Counter) Get() int   { return c.n }

// 指標 receiver — 修改呼叫者
func (c *Counter) Inc()      { c.n++ }

c := Counter{}
c.Inc()                    // 自動 &c
fmt.Println(c.Get())
Embedding(組合)
type Logger struct{ Prefix string }
func (l Logger) Log(msg string) {
    fmt.Println(l.Prefix, msg)
}

type Server struct {
    Logger              // 嵌入
    Addr string
}

s := Server{Logger: Logger{"[srv]"}}
s.Log("started")        // 升級到外層的方法
匿名結構
resp := struct {
    OK   bool
    Data []int
}{OK: true, Data: []int{1, 2, 3}}
原則:只要任何方法需要指標 receiver,就把所有方法都改成指標 receiver,以保一致。值型別的 method set 不包含指標 receiver 方法,可能導致無法滿足介面。

介面

介面是一組方法簽名。任何實作了這些方法的型別,都「隱式」滿足它 — 不需要 implements 關鍵字。

定義與滿足
type Stringer interface {
    String() string
}

type Point struct{ X, Y int }

func (p Point) String() string {
    return fmt.Sprintf("(%d,%d)", p.X, p.Y)
}

// Point 隱式滿足 Stringer
var s Stringer = Point{3, 4}
空介面 / any
// any === interface{}(Go 1.18+)
var v any = 42
v = "hello"
v = []int{1, 2}

// 型別斷言 — 取出具體型別
n, ok := v.(int)
if ok { fmt.Println("got int", n) }
標準函式庫常見介面
type error interface      { Error() string }
type Stringer interface   { String() string }
type Reader interface     { Read(p []byte) (n int, err error) }
type Writer interface     { Write(p []byte) (n int, err error) }
type Closer interface     { 關閉() error }
嵌入介面
type ReadCloser interface {
    Reader
    Closer
}
nil 介面 ≠ nil 指標:介面變數內含 (型別, 值) 對。如果把 nil *MyError 放進 error 變數,介面本身仍非 nil。if err != nil 會是 true,即便底層指標是 nil。不要把帶型別的 nil 當錯誤回傳。

Slices、Maps、陣列

陣列是固定大小的,在 Go 程式中很少用。Slice 是工作主力。Map 沒有順序。

Slice 基礎
var s []int                  // nil slice (len=0, cap=0)
s = []int{1, 2, 3}            // 字面值
s = make([]int, 5)             // len 5, cap 5
s = make([]int, 0, 10)           // len 0, cap 10

// re-slice(共享底層陣列!)
sub := s[1:3]
// 要複製: dst := make([]int, len(sub)); copy(dst, sub)
append(與成長機制)
s := []int{}
s = append(s, 1)           // 滿了通常 cap 會翻倍
s = append(s, 2, 3, 4)
s = append(s, other...)    // 合併另一個 slice

// 預先給足大小可避免 realloc
results := make([]int, 0, len(input))
Map
m := make(map[string]int)
m["a"] = 1
v, ok := m["a"]            // 存在則 ok==true
delete(m, "a")
n := len(m)

// 字面值
ages := map[string]int{
    "jane": 30,
    "alex": 28,
}

// 迭代順序是隨機的
for k, v := range ages { … }
陣列(固定大小)
var a [3]int            // [0 0 0]
b := [3]int{1, 2, 3}
c := [...]int{1, 2, 3}    // 大小由內容推斷

// 陣列是值 — 賦值會複製
d := b
d[0] = 99
// b 不變
slices.* 輔助函式(Go 1.21+)
import "slices"

slices.Sort(s)

slices.Index(s, x)
slices.Reverse(s)
slices.Equal(a, b)
slices.Concat(a, b)
maps.* 輔助函式(Go 1.21+)
import "maps"

keys := slices.Sorted(maps.Keys(m))
maps.複製(dst, src)
maps.DeleteFunc(m, func(k string, v int) bool {
    return v < 0

並行

不要透過共享記憶體來通訊;而要透過通訊來共享記憶體。Goroutine + 通道是 Go 的招牌賣點。

goroutine
go work(item)             // 在新的 goroutine 跑
go func() {
    fmt.Println("hi")
}()

// goroutine 很便宜(~2 KB stack)
// 執行期會把它們多工到 OS thread 上
channel
ch := make(chan int)         // 無緩衝
ch := make(chan int, 10)     // 有緩衝,容量 10

ch <- 42                     // 送(滿了/沒接收者就阻塞)
v := <-ch                    // 收
v, ok := <-ch                // 已關閉且空時 ok=false
close(ch)                    // 只能由送方關;接收方絕對不要關

// 直到關閉前持續 range
for v := range ch { fmt.Println(v) }
select
select {
case v := <-a:
    fmt.Println("a", v)
case v := <-b:
    fmt.Println("b", v)
case out <- 7:
    fmt.Println("sent")
case <-time.After(time.Second):
    fmt.Println("timeout")
default:
    fmt.Println("nothing ready")
}
sync.WaitGroup
var wg sync.WaitGroup
for _, item := range items {
    wg.Add(1)
    go func(it Item) {
        defer wg.完成()
        work(it)
    }(item)
}
wg.Wait()
sync.Mutex
type Counter struct {
    mu sync.Mutex
    n  int
}
func (c *Counter) Inc() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.n++
}
用 context 取消
ctx, cancel := context.WithTimeout(parent, 5*time.Second)
defer cancel()

select {
case <-ctx.完成():

    return v, nil

錯誤與泛型

錯誤是值。包它、檢查它,但不要用 panic 處理常規流程。

建立與包裹
errors.New("bad input")
fmt.Errorf("open %s: %w", path, err)   // %w 用來包裹

// 哨兵錯誤
var ErrNotFound = errors.New("not found")

// 自訂型別
type ParseError struct{ Line int; Msg string }
func (e *ParseError) Error() string {
    return fmt.Sprintf("line %d: %s", e.Line, e.Msg)
}
檢查:errors.Is / errors.As
if errors.Is(err, ErrNotFound) {
    // 透過包裹也能對到哨兵
}

var pe *ParseError
if errors.As(err, &pe) {
    fmt.Println("line", pe.Line)
}

// errors.Join (Go 1.20+)
err := errors.Join(err1, err2)
panic / recover
只在真的無法回復的 bug 使用(或最上層的 HTTP middleware)。
defer func() {
    if r := recover(); r != nil {
        log.Println("recovered:", r)

net/http 伺服器

Hello server
package main

import (
    "fmt"
    "net/http"
)

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /hello/{name}",
        func(w http.ResponseWriter, r *http.Request) {
            name := r.PathValue("name")
            fmt.Fprintf(w, "hi %s", name)
        })
    http.ListenAndServe(":8080", mux)
}

Middleware 模式
func withLog(h http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        h.ServeHTTP(w, r)
        slog.Info("req", "path", r.URL.Path,
                 "dur", time.Since(start))
    })
}

    for _, tc := range []struct {
        name    string
        a, b, w int

                t.Errorf("got %d, want %d", g, tc.w)

        })
    }
}

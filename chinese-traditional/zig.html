Hello, Zig

小巧、簡潔的語言,為穩固、最佳化、可重用軟體而生。沒有隱藏的控制流,沒有隱藏的記憶體配置。

標準程式

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

注意 !void 回傳型別 — main 可以回傳錯誤。try 會把錯誤往外傳。

建置與執行 terminal
# 編譯並執行單一檔案
zig run main.zig

# 只編譯成原生執行檔
zig build-exe main.zig

# 專案(使用 build.zig)
zig init
zig build run
zig build test

# 格式化你的程式碼
zig fmt main.zig

Zig 自帶用 Zig 寫的建置系統 — 不需要 Make,也不需要 CMake。

註解與測試 // /// test
// 一般註解
/// 文件註解(掛在下一個宣告上)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

測試直接寫在原始檔內。用 zig test file.zig 執行。

Zen of Zig philosophy
// 1. 精準傳達意圖。
// 2. 邊角情況很重要。
// 3. 寧可花時間讀程式,也不要花時間寫程式。
// 4. 一件事只該有一種顯而易見的做法。
// 5. 執行時 crash 比 bug 好。
// 6. 編譯期錯誤比執行時 crash 好。
// 7. 漸進改善。
// 8. 避開局部最佳解。
// 9. 減少需要記住的東西。
// 10. 把焦點放在程式碼,而不是風格。
// 11. 資源配置可能失敗;釋放必須成功。
// 12. 記憶體是一種資源。
// 13. 我們一起為使用者服務。

變數

兩種:const(不可變)與 var(可變)。所有宣告都必須使用 — 不允許死綁定。

const vs var 基礎
const name = "Ada";        // 不可變,推斷型別
var   counter: u32 = 0;  // 可變,顯式型別
counter += 1;

// 編譯錯誤: 不能對 const 賦值
// name = "Bob";

// 編譯錯誤: 沒用到的區域變數
// const x = 5;

真的非要丟棄,就用 _ = value;。

comptime 常數 comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// 編譯期求值
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

頂層 const 隱式為 comptime。

undefined 選擇性的不安全
var buf: [256]u8 = undefined;
// debug build 中 undefined 會填 0xAA bytes,
// 用來凸顯讀到未初始化的記憶體。
buf[0] = 42;
型別推斷 @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

數值與型別

Zig 有顯式、固定寬度的整數,並把溢位語意攤在陽光下。轉型永遠是顯式的。

整數型別

型別位元範圍
i8 / u88-128..127 / 0..255
i16 / u1616有號 / 無號
i32 / u3232最常用的整數
i64 / u6464大整數
isize / usizeptr指標寬度;index 用 usize
i7 / u23任意任意寬度 — 連 u3 都行!
f16 / f32 / f64 / f128不一IEEE 浮點數

轉型與溢位

轉型 @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // 執行期會檢查
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as 用於 coercion(擴寬、comptime)
const e = @as(i64, 10);

@intCast 在 safe build 下若塞不下會 trap。

溢位運算子 +% +| 等
// 預設的 + 在 safe build 下溢位會 trap
var x: u8 = 250;

// wrap 算術
const w = x +% 10;   // wrap 成 4

// 飽和算術
const s = x +| 10;   // 卡在 255

// 用 @addWithOverflow 抓溢位
const r = @addWithOverflow(x, 10);
// r 是 tuple: { result, did_overflow }
數字字面值 可讀性
const million = 1_000_000;
const hex     = 0xff_ee;
const oct     = 0o755;
const bin     = 0b1010_1010;
const exp     = 1.5e3;
const hexf    = 0x1.fp10;  // 16 進位浮點
字元與位元組 u8 'A'
const letter: u8 = 'A';       // 65
const emoji  : u21 = '😀';     // 字元編碼點

// 字串是 []const u8(UTF-8 位元組)
const greeting: []const u8 = "hi";

控制流

處處皆 expression。if、switch、區塊都會求值。switch 必須涵蓋所有情況。

if 即表達式 if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// 可選值 payload 綁定
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// 帶可選值的 while
while (it.next()) |item| {
    process(item);
}
for — 現代語法 for
const items = [_]u8{ 10, 20, 30 };

// 走訪每個值
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// 範圍語法(Zig 0.11+)
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// 多參數 for 同時取兩者
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// 對列舉做 switch(若窮盡就不需要 else)
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
標籤區塊與帶值 break :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// 跳出巢狀迴圈
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer 清理
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // 在離開 scope 時跑

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // 只在錯誤路徑跑

try initialize(buf);
return buf;          // 沒錯,就不會觸發 errdefer

函式

參數不可變。錯誤透過錯誤聯合回傳型別 !T 傳遞。

基本函式 fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// 參數是 const — 不能對 a 或 b 重新賦值
錯誤聯合返回 !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // 往外傳
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
用結構回傳多值 .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
函式指標 *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

錯誤

錯誤是值,但是特殊的一種。它們組成 set,透過 try 往外傳。

錯誤集合 error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

錯誤集合是命名錯誤的聯集。!T 用推斷的集合。

try 與 catch 控制
// try 會解開,出錯時直接回傳錯誤
const n = try parse(input);

// catch 用備用值取代
const m = parse(input) catch 0;

// catch 搭配 payload 綁定
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer 出錯時的清理
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // 失敗時 errdefer 會跑
    return t;               // 成功:跳過 errdefer
}
unreachable 斷言
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // safe build 下會 trap
}

// 「我知道這裡 parse 一定會成功」
const n = parse("42") catch unreachable;

記憶體與分配器

沒有隱藏的記憶體配置。每個會配置記憶體的函式都得收一個 Allocator。配置的東西要自己 free(或用 arena)。

分配器型樣

GeneralPurposeAllocator 適合除錯
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA 會偵測 leak 與重複釋放。應用與測試都該用它。

ArenaAllocator 批次釋放
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // 一次釋放所有東西
const alloc = arena.allocator();

// 隨意 alloc — 不必 free()
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
其他分配器 挑你愛的
// std.heap.page_allocator       — 以 mmap 為基底
// std.heap.c_allocator          — libc malloc
// std.testing.allocator         — 測試用
// std.heap.FixedBufferAllocator — 由 stack buffer 支援

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
陣列 / 切片 / 指標 [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // 固定大小
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // ptr + len
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // 指向單一項目的指標
const many: [*]const u8 = arr[0..].ptr; // 多項目指標

// slice 知道自己的長度: arr.len

結構與列舉

  
結構裝資料和方法。列舉可以攜帶 payload(tagged union)。兩者都是一等型別。


  

    

      

        
帶方法的結構 struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });

Comptime

  
Zig 的招牌功能。在編譯期執行程式碼、把型別當參數傳、生成程式碼 — 全部用普通的 Zig 語法。


  

    

      

        
用 comptime 型別做泛型 comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
每次呼叫都會生成一份新的特化版本 — 沒有執行期成本。

    


    

      

        
泛型結構 會回傳型別的 fn

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for 展開的迴圈

        
      

      
// 迴圈本體會在編譯期被每個 iteration 複製
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// 範圍版本也行
inline for (0..4) |i| {
    process(i); // 展開 4 次
}


    


    

      

        
在編譯期執行程式碼 comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// 在編譯期計算:
const tenth = comptime fib(10);  // 55,直接 baked in

建置系統

  
build.zig 就是 Zig 程式。你描述一張建置步驟的圖,編譯器會去執行。


  

    

      

        
最簡 build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

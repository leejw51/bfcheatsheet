Hello, Zig

Malý a jednoduchý jazyk navržený pro robustní, optimální a opakovaně použitelný software. Žádné skryté řízení toku, žádné skryté alokace.

Kanonický program

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

Všimněte si návratu !void — main může vracet chyby. try je propaguje.

Sestavit a spustit terminál
# Zkompiluje + spustí jeden soubor
zig run main.zig

# Pouze přeloží do nativního exe
zig build-exe main.zig

# Projekt (přes build.zig)
zig init
zig build run
zig build test

# Zformátuje váš kód
zig fmt main.zig

Zig má vlastní sestavovací systém napsaný v Zigu — žádný Make, žádný CMake.

Komentáře a testy // /// test
// běžný komentář
/// dokumentační komentář (žije nad další deklarací)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

Testy žijí přímo ve zdrojových souborech. Spustíte je zig test soubor.zig.

Proměnné

Dvě varianty: const (neměnná) a var (měnitelná). Všechny deklarace musí být použité — žádná mrtvá vazba.

const vs var základy
const name = "Ada";        // neměnná, typ odvozený
var   counter: u32 = 0;  // měnitelná, explicitní typ
counter += 1;

// Chyba překladu: přiřazení do const
// name = "Bob";

// Chyba překladu: nepoužitá lokální
// const x = 5;

Pokud opravdu musíte zahodit hodnotu, použijte _ = value;.

comptime konstanty comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// Vyhodnoceno v době kompilace
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

Top-level consty jsou implicitně comptime.

undefined opt-in nebezpečí
var buf: [256]u8 = undefined;
// V debug sestavení je undefined plnění bajty 0xAA,
// aby vyplavalo čtení neinicializovaných hodnot.
buf[0] = 42;
Odvození typu @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

Přetypování a přetečení

Přetypování @as @intCast

Řízení toku

Výrazy všude. if, switch, bloky — všechno se vyhodnocuje na hodnotu. switch musí být vyčerpávající.

if jako výraz if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// Vázání payloadu z optional
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// while s optional
while (it.next()) |item| {
    process(item);
}
for — moderní syntaxe for
const items = [_]u8{ 10, 20, 30 };

// iterace přes hodnoty
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// rozsahová syntaxe (Zig 0.11+)
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// obojí najednou s víceargumentovým for
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// Switch nad výčtem (žádné else, pokud je vyčerpávající)
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
Označené bloky a break s hodnotou :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// Vyskočení z vnořených smyček
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer úklid
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // proběhne při opuštění scope

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // pouze na chybové cestě

try initialize(buf);
return buf;          // errdefer se nespustí

Funkce

Parametry jsou neměnné. Chyby cestují přes návratové typy chybové unie !T.

Základní funkce fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// Parametry jsou const — nemůžete znovu přiřadit a ani b
Návrat chybové unie !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // propaguje
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
Více návratů přes strukturu .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
Ukazatel na funkci *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

Chyby

Chyby jsou hodnoty, ale zvláštního druhu. Tvoří množiny a propagují se přes try.

Chybové množiny error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

Chybové množiny jsou unie chyb pojmenovaných identifikátory. !T používá odvozenou množinu.

try a catch řízení
// try rozbalí nebo vrátí chybu
const n = try parse(input);

// catch dosadí hodnotu
const m = parse(input) catch 0;

// catch s vázáním payloadu
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer úklid při chybě
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // pokud selže, errdefer se spustí
    return t;               // úspěch: errdefer se přeskočí
}
unreachable tvrzení
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // past v bezpečných sestaveních
}

// "Vím, že tady parse uspěje"
const n = parse("42") catch unreachable;

Paměť a alokátory

Žádné skryté alokace. Každá funkce, která alokuje, přijímá Allocator. Uvolníte, co jste alokovali (nebo použijete arénu).

Vzory alokátorů

GeneralPurposeAllocator přátelský k ladění
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA detekuje úniky a dvojí uvolnění. Používejte v aplikacích a testech.

ArenaAllocator hromadné uvolnění
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // uvolní vše naráz
const alloc = arena.allocator();

// Alokujte volně — nikdy nevolejte free()
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
Další alokátory vyberte si jed
// std.heap.page_allocator       — postavený na mmap
// std.heap.c_allocator          — libc malloc
// std.testing.allocator         — pro testy
// std.heap.FixedBufferAllocator — podpořený stack bufferem

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
Pole / řezy / ukazatele [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // pevná velikost
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // ptr + délka
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // ukazatel na jednu položku
const many: [*]const u8 = arr[0..].ptr; // na více položek

// Řezy znají svou délku: arr.len

Struktury a výčty

  
Struktury drží data a metody. Výčty mohou nést payloady (značkované unie). Oboje jsou plnohodnotné typy.


  

    

      

        
Struktura s metodami struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });

Comptime

  
Zabijácká funkce Zigu. Spouštějte kód v době kompilace, berte typy jako parametry, generujte kód — vše v obyčejné syntaxi Zigu.


  

    

      

        
Generika přes typ comptime comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
Každé volání generuje čerstvou specializaci — bez nákladu za běhu.

    


    

      

        
Generická struktura funkce vracející typ

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for rozvinutá smyčka

        
      

      
// Tělo smyčky se v comptime duplikuje pro každou iteraci
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// Funguje i s rozsahem
inline for (0..4) |i| {
    process(i); // rozvinuto 4x
}


    


    

      

        
Spustit kód v době kompilace comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// Spočítáno během kompilace:
const tenth = comptime fib(10);  // 55, zapečené dovnitř

Sestavovací systém

  
build.zig je prostě Zig. Popisujete graf sestavovacích kroků; překladač je spouští.


  

    

      

        
Minimální build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

Bonjour, Zig

Un langage petit et simple conçu pour des logiciels robustes, optimaux et réutilisables. Pas de flux de contrôle caché, pas d'allocations cachées.

Le programme canonique

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

Notez le retour !void — main peut retourner des erreurs. try les propage.

Compiler et exécuter terminal
# Compiler + exécuter un seul fichier
zig run main.zig

# Compiler en exe natif uniquement
zig build-exe main.zig

# Projet (avec build.zig)
zig init
zig build run
zig build test

# Formater votre code
zig fmt main.zig

Zig possède son propre système de build écrit en Zig — pas de Make, pas de CMake.

Commentaires et tests // /// test
// commentaire normal
/// commentaire de doc (rattaché à la déclaration suivante)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

Les tests vivent dans les fichiers source. Lancez avec zig test file.zig.

Le Zen de Zig philosophie
// 1. Communiquer l'intention avec précision.
// 2. Les cas limites comptent.
// 3. Privilégier la lecture du code à son écriture.
// 4. Une seule manière évidente de faire les choses.
// 5. Les plantages à l'exécution valent mieux que les bugs.
// 6. Les erreurs de compilation valent mieux que les plantages à l'exécution.
// 7. Améliorations incrémentales.
// 8. Éviter les maximums locaux.
// 9. Réduire ce qu'il faut mémoriser.
// 10. Se concentrer sur le code plutôt que sur le style.
// 11. L'allocation de ressource peut échouer ; la libération doit réussir.
// 12. La mémoire est une ressource.
// 13. Ensemble nous servons les utilisateurs.

Variables

Deux saveurs : const (immuable) et var (mutable). Toutes les déclarations doivent être utilisées — pas de liaisons mortes.

const vs var bases
const name = "Ada";        // immuable, type inféré
var   counter: u32 = 0;  // mutable, type explicite
counter += 1;

// Erreur de compilation : affectation à const
// name = "Bob";

// Erreur de compilation : variable locale non utilisée
// const x = 5;

Utilisez _ = value; pour rejeter si vraiment nécessaire.

constantes comptime comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// Évalué à la compilation
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

Les const au niveau supérieur sont implicitement comptime.

undefined non-sûreté volontaire
var buf: [256]u8 = undefined;
// En build debug, undefined vaut 0xAA octets
// pour faire surgir les lectures non initialisées.
buf[0] = 42;
Inférence de type @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

Casts et dépassement

Casting @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // vérifié à l'exécution
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as sert à la coercition (élargissement, comptime)
const e = @as(i64, 10);

@intCast piège dans les builds sûrs si la valeur ne tient pas.

Opérateurs de dépassement +% +| etc
// Le + par défaut piège sur dépassement en build sûr
var x: u8 = 250;

// Arithmétique avec wrap
const w = x +% 10;   // se rabat à 4

// Arithmétique saturante
const s = x +| 10;   // plafonné à 255

// Capturer le dépassement avec @addWithOverflow
const r = @addWithOverflow(x, 10);
// r est un tuple : { result, did_overflow }
Littéraux numériques lisibilité
const million = 1_000_000;
const hex     = 0xff_ee;
const oct     = 0o755;
const bin     = 0b1010_1010;
const exp     = 1.5e3;
const hexf    = 0x1.fp10;  // flottant hex
Caractères et octets u8 'A'
const letter: u8 = 'A';       // 65
const emoji  : u21 = '😀';     // point de code

// Les chaînes sont []const u8 (octets UTF-8)
const greeting: []const u8 = "hi";

Flux de contrôle

Des expressions partout. if, switch, les blocs s'évaluent tous en valeurs. switch doit être exhaustif.

if comme expression if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// Liaison de payload optionnel
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// while avec optionnel
while (it.next()) |item| {
    process(item);
}
for — syntaxe moderne for
const items = [_]u8{ 10, 20, 30 };

// itérer les valeurs
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// syntaxe d'intervalle (Zig 0.11+)
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// les deux à la fois avec for multi-arg
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// Switch sur enum (pas de else si exhaustif)
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
Blocs étiquetés et break avec valeur :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// Sortir de boucles imbriquées
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer nettoyage
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // s'exécute à la sortie de portée

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // uniquement sur le chemin d'erreur

try initialize(buf);
return buf;          // errdefer ne se déclenche pas

Fonctions

Les paramètres sont immuables. Les erreurs voyagent via les types de retour union d'erreur !T.

Fonction de base fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// Les paramètres sont const — vous ne pouvez pas réaffecter a ou b
Retour avec union d'erreur !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // propager
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
Retours multiples via struct .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
Pointeur de fonction *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

Erreurs

Les erreurs sont des valeurs, mais d'un genre particulier. Elles forment des ensembles et se propagent via try.

Ensembles d'erreurs error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

Les ensembles d'erreurs sont des unions d'erreurs nommées. !T utilise un ensemble inféré.

try et catch contrôle
// try déballe ou retourne l'erreur
const n = try parse(input);

// catch substitue une valeur
const m = parse(input) catch 0;

// catch avec liaison du payload
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer nettoyage en cas d'erreur
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // si ça échoue, errdefer s'exécute
    return t;               // succès : errdefer est ignoré
}
unreachable assertion
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // piège dans les builds sûrs
}

// « Je sais que le parsing réussit ici »
const n = parse("42") catch unreachable;

Mémoire et allocateurs

Aucune allocation cachée. Chaque fonction qui alloue prend un Allocator. Vous libérez ce que vous allouez (ou utilisez une arène).

Patrons d'allocateurs

GeneralPurposeAllocator utile en debug
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

Le GPA détecte les fuites et les double-free. Utilisez-le dans les apps et les tests.

ArenaAllocator libération en bloc
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // libère tout d'un coup
const alloc = arena.allocator();

// Allouez librement — n'appelez jamais free()
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
Autres allocateurs choisissez votre poison
// std.heap.page_allocator       — basé sur mmap
// std.heap.c_allocator          — malloc de la libc
// std.testing.allocator         — pour les tests
// std.heap.FixedBufferAllocator — adossé à un tampon de pile

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
Tableaux / slices / pointeurs [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // taille fixe
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // ptr + len
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // ptr d'un seul élément
const many: [*]const u8 = arr[0..].ptr; // multi-éléments

// Les slices connaissent leur longueur : arr.len

Structs et enums

  
Les structs portent des données et des méthodes. Les enums peuvent transporter des payloads (unions étiquetées). Les deux sont des types de première classe.


  

    

      

        
Struct avec méthodes struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });

Union étiquetée union(enum)

        
      

      
const Token = union(enum) {
    int: i64,
    float: f64,
    text: []const u8,
    eof,
};

const t: Token = .{ .int = 42 };

switch (t) {
    .int => |n| std.debug.print("int {d}\n", .{n}),
    .float => |f| std.debug.print("float {d}\n", .{f}),
    .text => |s| std.debug.print("text {s}\n", .{s}),
    .eof => std.debug.print("end\n", .{}),
}

Comptime

  
La fonctionnalité phare de Zig. Exécutez du code à la compilation, prenez des types en paramètres, générez du code — le tout en syntaxe Zig classique.


  

    

      

        
Générique via type comptime comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
Chaque appel génère une nouvelle spécialisation — aucun coût à l'exécution.

    


    

      

        
Struct générique fn retournant un type

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for boucle déroulée

        
      

      
// Le corps de la boucle est dupliqué par itération à comptime
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// Forme avec intervalle aussi
inline for (0..4) |i| {
    process(i); // déroulé 4 fois
}


    


    

      

        
Exécuter du code à la compilation comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// Calculé pendant la compilation :
const tenth = comptime fib(10);  // 55, intégré

Système de build

  
build.zig est juste du Zig. Vous décrivez un graphe d'étapes de build ; le compilateur les exécute.


  

    

      

        
build.zig minimal build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

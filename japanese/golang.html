// 無限ループ

落とし穴 (1.22 以前): 古い Go ではループ変数が反復で再利用されたため、クロージャやゴルーチンで捕捉すると毎回同じアドレスを束縛していました。Go 1.22 で修正され、各反復で新しい v が割り当てられます。古い Go を対象とする場合はループ内で v := v としてください。

func divide(a, b float64) (float64, error) {

defer
関数の return 時に実行されます。LIFO 順。引数は defer 登録時に評価されます。
func readFile(path string) (string, error) {
    f, err := os.Open(path)
    if err != nil { return "", err }
    defer f.Close()    // 出口で必ず実行

    b, err := io.ReadAll(f)
    return string(b), err
}

ポインタ

Go にはポインタはありますがポインタ演算はありません。存在意義は二つ: コピーを避けることと、呼ばれた側が呼び出し側の値を変更できるようにすることです。

アドレス取得とデリファレンス
x := 42
p := &x         // x へのポインタ
fmt.Println(*p)  // 42
*p = 100
fmt.Println(x)   // 100
値渡し vs ポインタ渡し
func grow(n int)    { n *= 2 }      // 効果なし
func growP(n *int) { *n *= 2 }    // 呼び出し元を変更

x := 3
grow(x)        // x はまだ 3
growP(&x)      // x は 6 に
new vs &T{}
// new(T) → ゼロ値の T を指す *T
p := new(int)        // 0 を指す *int
*p = 7

// 複合リテラル — より慣用的
u := &User{Name: "jane"}
ポインタを使うとき
使うべき場面: 値を変更する場合、大きな構造体 (コピー回避)、ポインタレシーバでインターフェースを満たす場合、オプショナル/nil センチネル。
避けるべき場面: 小さな不変構造体、スライス/マップ/チャネル (既に参照型)。
落とし穴: 古い Go では range ループ変数のアドレスを取ると毎回同じアドレスが返されました。Go 1.22+ では各反復で新しい変数になるため、range ループ内の &v は安全です。古いバージョンでは事前に v := v としてください。

構造体 & メソッド

構造体はデータを保持します。メソッドはレシーバ付きの関数です。埋め込みによりコンポジションが実現できます。

定義 & インスタンス化
type User struct {
    ID    int      `json:"id"`
    Name  string   `json:"name"`
    Email string   `json:"email,omitempty"`
}

u := User{ID: 1, Name: "jane"}
v := User{1, "jane", ""}   // 位置指定、壊れやすい
p := &User{Name: "alex"}     // 新しい構造体へのポインタ
メソッド (値レシーバ vs ポインタレシーバ)
type Counter struct{ n int }

// 値レシーバ — コピーに対して動作
func (c Counter) Get() int   { return c.n }

// ポインタレシーバ — 呼び出し元を変更
func (c *Counter) Inc()      { c.n++ }

c := Counter{}
c.Inc()                    // 自動で &c
fmt.Println(c.Get())
埋め込み (コンポジション)
type Logger struct{ Prefix string }
func (l Logger) Log(msg string) {
    fmt.Println(l.Prefix, msg)
}

type Server struct {
    Logger              // 埋め込み
    Addr string
}

s := Server{Logger: Logger{"[srv]"}}
s.Log("started")        // プロモートされたメソッド
無名構造体
resp := struct {
    OK   bool
    Data []int
}{OK: true, Data: []int{1, 2, 3}}
ルール: どれか一つでもポインタレシーバが必要なら、一貫性のためにすべてポインタレシーバにしましょう。値型のメソッドセットにはポインタレシーバのメソッドが含まれず、インターフェース充足が壊れることがあります。

インターフェース

インターフェースはメソッドシグネチャの集合です。それらをすべて実装する型は暗黙的にインターフェースを満たします。implements キーワードはありません。

定義と実装
type Stringer interface {
    String() string
}

type Point struct{ X, Y int }

func (p Point) String() string {
    return fmt.Sprintf("(%d,%d)", p.X, p.Y)
}

// Point は暗黙的に Stringer を満たす
var s Stringer = Point{3, 4}
空インターフェース / any
// any === interface{} (Go 1.18+)
var v any = 42
v = "hello"
v = []int{1, 2}

// 型アサーション — 具体型を取り出す
n, ok := v.(int)
if ok { fmt.Println("got int", n) }
よく使う標準ライブラリのインターフェース
type error interface      { Error() string }
type Stringer interface   { String() string }
type Reader interface     { Read(p []byte) (n int, err error) }
type Writer interface     { Write(p []byte) (n int, err error) }
type Closer interface     { Close() error }
インターフェースの埋め込み
type ReadCloser interface {
    Reader
    Closer
}
nil インターフェース ≠ nil ポインタ: インターフェース変数は (型, 値) を保持します。nil の *MyError を error 変数に格納すると、インターフェース自体は非 nil になります。if err != nil は基底ポインタが nil でも true になります。型付き nil をエラーとして返すのは避けましょう。

スライス, マップ, 配列

スライスの基本

// 再スライス (バッキング配列を共有!)

append (と容量の伸長)
s := []int{}

s = append(s, 2, 3, 4)
s = append(s, other...)    // 別のスライスを結合

// サイズが既知なら事前確保 — 再確保を回避
results := make([]int, 0, len(input))

ゴルーチン

チャネル

エラー & ジェネリクス

エラーは値です。ラップして検査しますが、通常のフローで panic を起こさないようにしましょう。

作成 & ラップ
errors.New("bad input")
fmt.Errorf("open %s: %w", path, err)   // %w でラップ

// センチネル
var ErrNotFound = errors.New("not found")

// カスタム型
type ParseError struct{ Line int; Msg string }
func (e *ParseError) Error() string {
    return fmt.Sprintf("line %d: %s", e.Line, e.Msg)
}
検査: errors.Is / errors.As
if errors.Is(err, ErrNotFound) {
    // ラップ越しでセンチネルにマッチ
}

var pe *ParseError
if errors.As(err, &pe) {
    fmt.Println("line", pe.Line)
}

// errors.Join (Go 1.20+)
err := errors.Join(err1, err2)

本当に回復不能なバグ (またはトップレベルの HTTP ミドルウェア) のときだけ使いましょう。

Hello サーバー
package main

import (
    "fmt"
    "net/http"
)

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /hello/{name}",
        func(w http.ResponseWriter, r *http.Request) {
            name := r.PathValue("name")
            fmt.Fprintf(w, "hi %s", name)
        })
    http.ListenAndServe(":8080", mux)
}

func withLog(h http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        h.ServeHTTP(w, r)
        slog.Info("req", "path", r.URL.Path,
                 "dur", time.Since(start))
    })
}

func TestAdd(t *testing.T) {
    for _, tc := range []struct {
        name    string
        a, b, w int
    }{
        {"zero", 0, 0, 0},
        {"pos",  2, 3, 5},
        {"neg", -1, 1, 0},
    } {
        t.Run(tc.name, func(t *testing.T) {
            if g := add(tc.a, tc.b); g != tc.w {
                t.Errorf("got %d, want %d", g, tc.w)
            }
        })
    }
}

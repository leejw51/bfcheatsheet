こんにちは、Zig

堅牢で最適で再利用可能なソフトウェアのために設計された小さくてシンプルな言語。隠れた制御フローや隠れたアロケーションがない。

正典のプログラム

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

戻り値の !void に注目 — main はエラーを返せる。try がそれを伝播する。

ビルドと実行 terminal
# 単一ファイルをコンパイル + 実行
zig run main.zig

# ネイティブ実行ファイルへコンパイルのみ
zig build-exe main.zig

# プロジェクト(build.zig 使用)
zig init
zig build run
zig build test

# コードをフォーマット
zig fmt main.zig

Zig は Zig で書かれた独自のビルドシステムを持つ — Make も CMake も不要。

コメントとテスト // /// test
// 通常のコメント
/// ドキュメントコメント(次の宣言に紐付く)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

テストはソースファイル内に書く。zig test file.zig で実行。

Zig の禅 philosophy
// 1. 意図を正確に伝える。
// 2. エッジケースが重要。
// 3. 書くより読むコードを優先する。
// 4. やり方は明白なひとつだけ。
// 5. 実行時クラッシュはバグよりまし。
// 6. コンパイルエラーは実行時クラッシュよりまし。
// 7. 漸進的な改善。
// 8. 局所最適を避ける。
// 9. 覚えるべき量を減らす。
// 10. スタイルよりコードに注力。
// 11. リソース確保は失敗してよいが、解放は成功しなければならない。
// 12. メモリはリソース。
// 13. 共にユーザーに尽くす。

変数

2 種類: const(不変)と var(可変)。すべての宣言は使われなければならない — 死蔵バインディング不可。

const と var 基本
const name = "Ada";        // 不変、型推論
var   counter: u32 = 0;  // 可変、明示型
counter += 1;

// コンパイルエラー: const への代入
// name = "Bob";

// コンパイルエラー: 未使用ローカル
// const x = 5;

どうしても破棄したいなら _ = value; を使う。

コンプタイム定数 comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// コンパイル時に評価
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

トップレベルの const は暗黙に comptime。

undefined 明示的な未初期化
var buf: [256]u8 = undefined;
// デバッグビルドでは undefined は 0xAA バイト、
// 未初期化読み出しを目立たせる。
buf[0] = 42;
型推論 @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

数値と型

Zig は明示的で固定幅の整数を持ち、オーバーフローのセマンティクスを表に出します。キャストは常に明示的。

整数型

型ビット幅範囲
i8 / u88-128..127 / 0..255
i16 / u1616符号付き / 符号なし
i32 / u3232もっとも一般的な整数
i64 / u6464大きな整数
isize / usizeポインタ幅ポインタ幅整数。インデックスには usize
i7 / u23任意任意ビット幅 — u3 も有効!
f16 / f32 / f64 / f128多様IEEE 浮動小数点

キャストとオーバーフロー

キャスト @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // 実行時チェック
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as は型強制(拡幅、コンプタイム)用
const e = @as(i64, 10);

@intCast は安全ビルドで値が収まらない場合トラップする。

オーバーフロー演算子 +% +| 等
// デフォルトの + は安全ビルドでオーバーフロー時にトラップ
var x: u8 = 250;

// ラップする算術
const w = x +% 10;   // 4 にラップ

// 飽和算術
const s = x +| 10;   // 255 で固定

// @addWithOverflow でオーバーフローを捕捉
const r = @addWithOverflow(x, 10);
// r はタプル: { result, did_overflow }
数値リテラル 可読性
const million = 1_000_000;
const hex     = 0xff_ee;
const oct     = 0o755;
const bin     = 0b1010_1010;
const exp     = 1.5e3;
const hexf    = 0x1.fp10;  // 16 進浮動小数
文字とバイト u8 'A'
const letter: u8 = 'A';       // 65
const emoji  : u21 = '😀';     // コードポイント

// 文字列は []const u8(UTF-8 バイト列)
const greeting: []const u8 = "hi";

制御フロー

あらゆる場所が式。if、switch、ブロックはすべて値に評価される。switch は網羅的でなければならない。

式としての if if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// オプショナルのペイロードバインド
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// while でオプショナル
while (it.next()) |item| {
    process(item);
}
for — モダン構文 for
const items = [_]u8{ 10, 20, 30 };

// 値を反復
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// 範囲構文(Zig 0.11+)
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// 多引数 for で両方を同時に
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// enum で switch(網羅していれば else 不要)
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
ラベル付きブロックと値を伴う break :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// 入れ子ループから抜ける
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer クリーンアップ
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // スコープ退出時に走る

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // エラー経路でのみ

try initialize(buf);
return buf;          // errdefer は発火しない

関数

パラメータは不変。エラーはエラーユニオン戻り値型 !T 経由で伝播する。

基本の関数 fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// パラメータは const — a や b に再代入できない
エラーユニオン戻り値 !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // 伝播
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
struct で複数戻り値 .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
関数ポインタ *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

エラー

エラーは値ですが、特別な種類です。集合を作り、try 経由で伝播します。

エラーセット error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

エラーセットは名前付きエラーの和。!T は推論されたセットを使う。

try と catch 制御
// try は値を取り出すかエラーを返す
const n = try parse(input);

// catch は値を差し替える
const m = parse(input) catch 0;

// catch でペイロードバインド
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer エラー時クリーンアップ
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // 失敗すれば errdefer が走る
    return t;               // 成功: errdefer はスキップ
}
unreachable アサーション
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // 安全ビルドではトラップ
}

// "ここではパース成功すると分かっている"
const n = parse("42") catch unreachable;

メモリとアロケータ

隠れたアロケーションは無し。アロケートする関数はすべて Allocator を受け取る。確保したものは解放する(あるいはアリーナを使う)。

アロケータパターン

GeneralPurposeAllocator デバッグに便利
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA はリークと二重 free を検出する。アプリやテストで使おう。

ArenaAllocator 一括解放
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // すべて一度に解放
const alloc = arena.allocator();

// 自由にアロケート — free() は呼ばない
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
他のアロケータ 毒を選べ
// std.heap.page_allocator       — mmap ベース
// std.heap.c_allocator          — libc malloc
// std.testing.allocator         — テスト用
// std.heap.FixedBufferAllocator — スタックバッファ裏付け

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
配列 / スライス / ポインタ [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // 固定長
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // ptr + len
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // 単要素ポインタ
const many: [*]const u8 = arr[0..].ptr; // 多要素

// スライスは長さを知っている: arr.len

構造体と列挙型

  
構造体はデータとメソッドを持ちます。列挙型はペイロードを持てます(タグ付き共用体)。両方ともファーストクラスの型です。


  

    

      

        
メソッド付き構造体 struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });


    


    

      

        
列挙型 enum

        
      

      
const Direction = enum { north, south, east, west };

    ok = 200,

    server_error = 500,

const code = @intFromEnum(Status.ok); // 200

    int: i64,
    float: f64,

    eof,

const t: Token = .{ .int = 42 };

switch (t) {
    .int => |n| std.debug.print("int {d}\n", .{n}),
    .float => |f| std.debug.print("float {d}\n", .{f}),
    .text => |s| std.debug.print("text {s}\n", .{s}),
    .eof => std.debug.print("end\n", .{}),
}

コンプタイム

  
Zig のキラー機能。コンパイル時にコードを実行し、型をパラメータとして取り、コードを生成 — すべて素の Zig 構文で。


  

    

      

        
コンプタイム型でジェネリック comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
呼び出しごとに新しい特殊化が生成される — 実行時コストなし。

    


    

      

        
ジェネリック構造体 型を返す関数

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for 展開ループ

        
      

      
// ループ本体はコンプタイムで反復ごとに複製される
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// 範囲形式も
inline for (0..4) |i| {
    process(i); // 4 回展開
}


    


    

      

        
コンパイル時にコードを実行 comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// コンパイル中に計算される:
const tenth = comptime fib(10);  // 55、埋め込み済み

ビルドシステム

  
build.zig はただの Zig。ビルドステップのグラフを記述すれば、コンパイラが実行します。


  

    

      

        
最小の build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

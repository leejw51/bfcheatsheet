Hello, Zig

견고하고, 최적이며, 재사용 가능한 소프트웨어를 위해 설계된 작고 간단한 언어. 숨겨진 제어 흐름 없음, 숨겨진 할당 없음.

표준 프로그램

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

반환 타입 !void에 주목 — main은 에러를 반환할 수 있습니다. try가 에러를 전파합니다.

빌드 & 실행 terminal
# 단일 파일 컴파일 + 실행
zig run main.zig

# 네이티브 실행 파일로 컴파일만
zig build-exe main.zig

# 프로젝트 (build.zig 사용)
zig init
zig build run
zig build test

# 코드 포매팅
zig fmt main.zig

Zig은 Zig으로 작성된 자체 빌드 시스템을 갖고 있습니다 — Make도 CMake도 필요 없습니다.

주석 & 테스트 // /// test
// 일반 주석
/// 문서 주석 (다음 선언에 부착됨)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

테스트는 소스 파일 안에 함께 둡니다. zig test file.zig로 실행하세요.

Zen of Zig philosophy
// 1. 의도를 정확히 전달하라.
// 2. 엣지 케이스가 중요하다.
// 3. 코드를 쓰는 것보다 읽는 것을 선호하라.
// 4. 한 가지를 하는 명백한 방법은 하나만.
// 5. 런타임 크래시가 버그보다 낫다.
// 6. 컴파일 에러가 런타임 크래시보다 낫다.
// 7. 점진적 개선.
// 8. 지역 최댓값을 피하라.
// 9. 기억해야 할 양을 줄여라.
// 10. 스타일보다 코드에 집중하라.
// 11. 자원 할당은 실패할 수 있지만, 해제는 반드시 성공해야 한다.
// 12. 메모리는 자원이다.
// 13. 우리 모두 함께 사용자를 섬긴다.

변수

두 가지 형태가 있습니다: const (불변)와 var (가변). 모든 선언은 사용되어야 합니다 — 죽은 바인딩은 허용되지 않습니다.

const vs var 기본
const name = "Ada";        // 불변, 타입 추론
var   counter: u32 = 0;  // 가변, 명시적 타입
counter += 1;

// 컴파일 에러: const에 재할당
// name = "Bob";

// 컴파일 에러: 사용되지 않은 지역 변수
// const x = 5;

정말 필요하다면 _ = value;로 값을 버릴 수 있습니다.

컴파일타임 상수 comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// 컴파일타임에 평가됨
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

최상위 const는 암묵적으로 comptime입니다.

undefined 선택적 비안전성
var buf: [256]u8 = undefined;
// 디버그 빌드에서 undefined는 0xAA 바이트로 채워져
// 초기화되지 않은 읽기를 드러냅니다.
buf[0] = 42;
타입 추론 @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

숫자 & 타입

Zig은 명시적인 고정 폭 정수를 갖고 있고 오버플로 의미를 그대로 노출합니다. 캐스팅은 항상 명시적입니다.

정수 타입

타입비트범위
i8 / u88-128..127 / 0..255
i16 / u1616부호 있음 / 부호 없음
i32 / u3232가장 흔히 쓰는 정수
i64 / u6464큰 정수
isize / usizeptr포인터 크기; 인덱스에는 usize
i7 / u23임의임의 폭 — u3도 유효합니다!
f16 / f32 / f64 / f128가변IEEE 부동소수점

캐스팅 & 오버플로

캐스팅 @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // 런타임 검사
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as는 강제 변환용 (확장, comptime)
const e = @as(i64, 10);

값이 들어맞지 않으면 @intCast는 안전 빌드에서 트랩을 발생시킵니다.

오버플로 연산자 +% +| 등
// 기본 +는 안전 빌드에서 오버플로 시 트랩
var x: u8 = 250;

// 래핑 산술
const w = x +% 10;   // 4로 감싸짐

// 포화 산술
const s = x +| 10;   // 255에 고정

// @addWithOverflow로 오버플로를 잡아내기
const r = @addWithOverflow(x, 10);
// r은 튜플: { result, did_overflow }

함수

매개변수는 불변입니다. 에러는 에러 유니온 반환 타입 !T를 통해 전달됩니다.

기본 함수 fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// 매개변수는 const — a나 b를 재할당할 수 없습니다
에러 유니온 반환 !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // 에러 전파
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
struct로 다중 반환 .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
함수 포인터 *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

에러

에러는 값이지만 특별한 종류의 값입니다. 집합을 이루며 try를 통해 전파됩니다.

에러 집합 error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

에러 집합은 명명된 에러들의 유니온입니다. !T는 추론된 집합을 사용합니다.

try와 catch 제어
// try는 값을 풀거나 에러를 반환
const n = try parse(input);

// catch는 대체 값을 제공
const m = parse(input) catch 0;

// 페이로드 바인딩이 있는 catch
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer 에러 시 정리
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // 실패 시 errdefer 실행
    return t;               // 성공: errdefer는 건너뜀
}
unreachable 단언
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // 안전 빌드에서는 트랩
}

// "여기서 파싱이 성공함을 안다"
const n = parse("42") catch unreachable;

메모리 & 할당자

숨겨진 할당이 없습니다. 할당하는 모든 함수는 Allocator를 받습니다. 할당한 만큼 직접 해제하거나(아니면 아레나를 사용하세요).

할당자 패턴

GeneralPurposeAllocator 디버그 친화
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA는 누수와 이중 해제를 감지합니다. 앱과 테스트에서 사용하세요.

ArenaAllocator 일괄 해제
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // 모든 할당을 한 번에 해제
const alloc = arena.allocator();

// 자유롭게 할당 — free()는 절대 호출하지 않음
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
기타 할당자 취향대로 고르세요
// std.heap.page_allocator       — mmap 기반
// std.heap.c_allocator          — libc malloc
// std.testing.allocator         — 테스트용
// std.heap.FixedBufferAllocator — 스택 버퍼 기반

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
배열 / 슬라이스 / 포인터 [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // 고정 크기
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // 포인터 + 길이
const sub  : []const u8 = arr[0..2];

const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });

Comptime

  
Zig의 킬러 기능. 컴파일타임에 코드를 실행하고, 타입을 매개변수로 받고, 코드를 생성합니다 — 모두 평범한 Zig 문법으로요.


  

    

      

        
컴파일타임 타입을 통한 제네릭 comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
호출마다 새로운 특화가 생성됩니다 — 런타임 비용은 없습니다.

    


    

      

        
제네릭 구조체 타입을 반환하는 fn

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;

빌드 시스템

  
build.zig는 그저 Zig 코드입니다. 빌드 단계의 그래프를 기술하면 컴파일러가 그것을 실행합니다.


  

    

      

        
최소한의 build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}

Hello, Zig

A small, simple language designed for robust, optimal, and reusable software. No hidden control flow, no hidden allocations.

The Canonical Program

Hello, World main.zig
const std = @import("std");

pub fn main() !void {
    const stdout = std.io.getStdOut().writer();
    try stdout.print("Hello, {s}!\n", .{"Zig"});
}

Note the !void return — main can return errors. try propagates them.

Build & Run terminal
# Compile + run a single file
zig run main.zig

# Just compile to native exe
zig build-exe main.zig

# Project (using build.zig)
zig init
zig build run
zig build test

# Format your code
zig fmt main.zig

Zig has its own build system written in Zig — no Make, no CMake.

Comments & Tests // /// test
// regular comment
/// doc comment (lives on next decl)
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

test "add works" {
    try std.testing.expectEqual(5, add(2, 3));
}

Tests live in source files. Run with zig test file.zig.

Zen of Zig philosophy
// 1. Communicate intent precisely.
// 2. Edge cases matter.
// 3. Favor reading code over writing code.
// 4. Only one obvious way to do things.
// 5. Runtime crashes are better than bugs.
// 6. Compile errors are better than runtime crashes.
// 7. Incremental improvements.
// 8. Avoid local maximums.
// 9. Reduce the amount one must remember.
// 10. Focus on code rather than style.
// 11. Resource allocation may fail; deallocation must succeed.
// 12. Memory is a resource.
// 13. Together we serve the users.

Variables

Two flavors: const (immutable) and var (mutable). All declarations must be used — no dead bindings.

const vs var basics
const name = "Ada";        // immutable, type inferred
var   counter: u32 = 0;  // mutable, explicit type
counter += 1;

// Compile error: assignment to const
// name = "Bob";

// Compile error: unused local
// const x = 5;

Use _ = value; to discard if you really must.

comptime constants comptime
const PI: f64 = 3.14159;
const MAX_USERS = 1024;

// Compile-time evaluated
comptime {
    const sum = 2 + 3;
    @compileLog("sum is ", sum);
}

Top-level consts are implicitly comptime.

undefined opt-in unsafety
var buf: [256]u8 = undefined;
// In debug builds, undefined is 0xAA bytes
// to surface uninitialized reads.
buf[0] = 42;
Type inference @TypeOf
const x = 42;             // comptime_int
const y: i32 = 42;       // i32
const z = @as(i32, 42); // i32

const T = @TypeOf(y); // T == i32

Casts & Overflow

Casting @as @intCast
const a: i32 = 300;
const b: u8 = @intCast(a);   // runtime checked
const c: f32 = @floatFromInt(a);
const d: i32 = @intFromFloat(3.7);

// @as is for coercion (widening, comptime)
const e = @as(i64, 10);

@intCast traps in safe builds if the value doesn't fit.

Overflow operators +% +| etc
// Default + traps on overflow in safe builds
var x: u8 = 250;

// Wrapping arithmetic
const w = x +% 10;   // wraps to 4

// Saturating arithmetic
const s = x +| 10;   // pinned to 255

// Catch overflow with @addWithOverflow
const r = @addWithOverflow(x, 10);
// r is a tuple: { result, did_overflow }
Number literals readability
const million = 1_000_000;
const hex     = 0xff_ee;
const oct     = 0o755;
const bin     = 0b1010_1010;
const exp     = 1.5e3;
const hexf    = 0x1.fp10;  // hex float
Characters & bytes u8 'A'
const letter: u8 = 'A';       // 65
const emoji  : u21 = '😀';     // codepoint

// Strings are []const u8 (UTF-8 bytes)
const greeting: []const u8 = "hi";

Control Flow

Expressions everywhere. if, switch, blocks all evaluate to values. switch must be exhaustive.

if as expression if
const x = 10;
const sign = if (x > 0) "+" else if (x < 0) "-" else "0";

// Optional payload binding
const maybe: ?i32 = 42;
if (maybe) |val| {
    std.debug.print("got {d}\n", .{val});
} else {
    std.debug.print("null\n", .{});
}
while while
var i: u32 = 0;
while (i < 10) : (i += 1) {
    if (i == 3) continue;
    if (i == 7) break;
    std.debug.print("{d}\n", .{i});
}

// while with optional
while (it.next()) |item| {
    process(item);
}
for — modern syntax for
const items = [_]u8{ 10, 20, 30 };

// iterate values
for (items) |it| {
    std.debug.print("{d}\n", .{it});
}

// range syntax (Zig 0.11+)
for (0..items.len) |i| {
    std.debug.print("i={d}\n", .{i});
}

// both at once with multi-arg for
for (items, 0..) |it, i| {
    std.debug.print("{d}: {d}\n", .{ i, it });
}
switch switch
const n: u8 = 3;
const name = switch (n) {
    0 => "zero",
    1, 2, 3 => "few",
    4...9 => "some",
    else => "many",
};

// Switch on enum (no else needed if exhaustive)
const Color = enum { red, green, blue };
const hex: []const u8 = switch (color) {
    .red => "#f00",
    .green => "#0f0",
    .blue => "#00f",
};
Labeled blocks & break with value :label
const result = blk: {
    var sum: u32 = 0;
    for (items) |x| sum += x;
    break :blk sum * 2;
};

// Break out of nested loops
outer: for (rows) |row| {
    for (row) |cell| {
        if (cell == target) break :outer;
    }
}
defer / errdefer cleanup
const file = try std.fs.cwd().openFile("x", .{});
defer file.close();   // runs at scope exit

const buf = try alloc.alloc(u8, 100);
errdefer alloc.free(buf);  // only on error path

try initialize(buf);
return buf;          // no errdefer fires

Functions

Parameters are immutable. Errors travel via error union return types !T.

Basic function fn
pub fn add(a: i32, b: i32) i32 {
    return a + b;
}

// Parameters are const — you can't reassign a or b
Error union return !T
fn parse(s: []const u8) !i32 {
    return std.fmt.parseInt(i32, s, 10);
}

pub fn main() !void {
    const n = try parse("42");  // propagate
    const m = parse("oops") catch 0;
    _ = .{ n, m };
}
Multiple returns via struct .{}
fn divmod(a: u32, b: u32) struct { q: u32, r: u32 } {
    return .{ .q = a / b, .r = a % b };
}

const result = divmod(17, 5);
// result.q == 3, result.r == 2
Function pointer *const fn
const Op = *const fn(i32, i32) i32;

fn apply(op: Op, a: i32, b: i32) i32 {
    return op(a, b);
}

// const r = apply(add, 2, 3); // 5

Errors

Errors are values, but a special kind. They form sets and propagate via try.

Error sets error{...}
const ParseError = error{
    InvalidCharacter,
    Overflow,
    Empty,
};

fn demo() ParseError!i32 {
    return error.Empty;
}

Error sets are unions of named errors. !T uses an inferred set.

try and catch control
// try unwraps or returns the error
const n = try parse(input);

// catch substitutes a value
const m = parse(input) catch 0;

// catch with payload binding
const p = parse(input) catch |err| switch (err) {
    error.Overflow => 9999,
    else => return err,
};
errdefer on-error cleanup
fn build(alloc: std.mem.Allocator) !*Thing {
    const t = try alloc.create(Thing);
    errdefer alloc.destroy(t);

    try t.init();           // if this fails, errdefer runs
    return t;               // success: errdefer is skipped
}
unreachable assertion
switch (variant) {
    .a => doA(),
    .b => doB(),
    .c => unreachable,  // trap in safe builds
}

// "I know parsing succeeds here"
const n = parse("42") catch unreachable;

Memory & Allocators

No hidden allocations. Every function that allocates takes an Allocator. You free what you allocate (or use an arena).

Allocator Patterns

GeneralPurposeAllocator debug-friendly
var gpa = std.heap.GeneralPurposeAllocator(.{}){};
defer _ = gpa.deinit();
const alloc = gpa.allocator();

const buf = try alloc.alloc(u8, 128);
defer alloc.free(buf);

GPA detects leaks and double-frees. Use it in apps and tests.

ArenaAllocator bulk free
var arena = std.heap.ArenaAllocator.init(parent_alloc);
defer arena.deinit();   // frees everything at once
const alloc = arena.allocator();

// Allocate freely — never call free()
const a = try alloc.alloc(u8, 10);
const b = try alloc.alloc(u8, 20);
_ = .{ a, b };
Other allocators choose your poison
// std.heap.page_allocator       — mmap-based
// std.heap.c_allocator          — libc malloc
// std.testing.allocator         — for tests
// std.heap.FixedBufferAllocator — backed by a stack buffer

var buf: [1024]u8 = undefined;
var fba = std.heap.FixedBufferAllocator.init(&buf);
const alloc = fba.allocator();
Arrays / slices / pointers [N]T [_]T []T *T
const arr: [3]u8 = .{ 1, 2, 3 };  // fixed-size
const inferred = [_]u8{ 1, 2, 3 };

const slice: []const u8 = arr[0..]; // ptr + len
const sub  : []const u8 = arr[0..2];

const p: *const u8 = &arr[0];      // single-item ptr
const many: [*]const u8 = arr[0..].ptr; // many-item

// Slices know their length: arr.len

Structs & Enums

  
Structs hold data and methods. Enums can carry payloads (tagged unions). Both are first-class types.


  

    

      

        
Struct with methods struct

        
      

      
const Point = struct {
    x: f32,
    y: f32,

    pub fn init(x: f32, y: f32) Point {
        return .{ .x = x, .y = y };
    }

    pub fn distance(self: Point, other: Point) f32 {
        const dx = self.x - other.x;
        const dy = self.y - other.y;
        return @sqrt(dx*dx + dy*dy);
    }
};

const p = Point.init(1, 2);
const d = p.distance(.{ .x = 3, .y = 4 });


    


    

      

        
Enum enum

        
      

      
const Direction = enum { north, south, east, west };

// Specify tag type and explicit values
const Status = enum(u8) {
    ok = 200,
    not_found = 404,
    server_error = 500,
};

const code = @intFromEnum(Status.ok); // 200


    


    

      

        
Tagged union union(enum)

        
      

      
const Token = union(enum) {
    int: i64,
    float: f64,
    text: []const u8,
    eof,
};

const t: Token = .{ .int = 42 };

switch (t) {
    .int => |n| std.debug.print("int {d}\n", .{n}),
    .float => |f| std.debug.print("float {d}\n", .{f}),
    .text => |s| std.debug.print("text {s}\n", .{s}),
    .eof => std.debug.print("end\n", .{}),
}


    


    

      

        
Anonymous struct literals .{}

        
      

      
// Type inferred from context
const p: Point = .{ .x = 1, .y = 2 };

// Tuple-style (positional)
const tup = .{ 42, "hi", true };
// access: tup[0], tup[1], tup[2]

// Used pervasively for print args
std.debug.print("{d} {s}\n", .{ 1, "hi" });


    


    

      

        
packed struct bitfields

        
      

      
const Flags = packed struct {
    read:  bool,
    write: bool,
    exec:  bool,
    _pad:  u5,
};
// @sizeOf(Flags) == 1 byte

const f: Flags = .{ .read = true, .write = false, .exec = true, ._pad = 0 };


    


    

      

        
Optionals ?T

        
      

      
var maybe: ?i32 = null;
maybe = 42;

if (maybe) |v| {
    std.debug.print("got {d}\n", .{v});
}

// .? unwraps or panics
const n = maybe.?;

// orelse provides default
const m = maybe orelse 0;


    

  







  
Comptime

  
Zig's killer feature. Run code at compile time, take types as parameters, generate code — all in plain Zig syntax.


  

    

      

        
Generic via comptime type comptime T

        
      

      
fn max(comptime T: type, a: T, b: T) T {
    return if (a > b) a else b;
}

const x = max(i32, 3, 7);    // 7
const y = max(f64, 1.5, 2.5); // 2.5


      
Each call generates a fresh specialization — no runtime cost.

    


    

      

        
Generic struct type-returning fn

        
      

      
fn List(comptime T: type) type {
    return struct {
        items: []T,
        len: usize,
    };
}

const IntList = List(i32);
var xs: IntList = .{ .items = &[_]i32{}, .len = 0 };
_ = xs;


    


    

      

        
inline for unrolled loop

        
      

      
// Loop body is duplicated per iteration at comptime
const fields = @typeInfo(MyStruct).Struct.fields;
inline for (fields) |f| {
    std.debug.print("field: {s}\n", .{f.name});
}

// Range form too
inline for (0..4) |i| {
    process(i); // expanded 4 times
}


    


    

      

        
Run code at compile time comptime { }

        
      

      
fn fib(n: u32) u64 {
    return if (n < 2) n else fib(n - 1) + fib(n - 2);
}

// Computed during compilation:
const tenth = comptime fib(10);  // 55, baked in

Build System

  
build.zig is just Zig. You describe a graph of build steps; the compiler runs them.


  

    

      

        
Minimal build.zig build.zig

        
      

      
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const exe = b.addExecutable(.{
        .name = "hello",
        .root_source_file = b.path("src/main.zig"),
        .target = target,
        .optimize = optimize,
    });
    b.installArtifact(exe);

    const run = b.addRunArtifact(exe);
    const run_step = b.step("run", "Run the app");
    run_step.dependOn(&run.step);
}


    


    

      

        
Common commands cli

        
      

      
zig build              # default install step
zig build run          # custom run step
zig build test         # run tests
zig build -Doptimize=ReleaseFast
zig build -Dtarget=x86_64-linux-gnu

zig build --help       # list user-defined steps/options


    


    

      

        
Optimize modes -Doptimize

        
      

      
// Debug      — default; safety on, no optimization
// ReleaseSafe — optimized + runtime safety checks
// ReleaseFast — optimized, no safety checks
// ReleaseSmall — size-optimized

zig build -Doptimize=ReleaseSafe
